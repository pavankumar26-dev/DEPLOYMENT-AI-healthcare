// Simple demo functionality
function runDemo() {
const age = document.getElementById('age').value;
const gender = document.getElementById('gender').value;
const bmi = document.getElementById('bmi').value;
const bp = document.getElementById('blood-pressure').value;
const cholesterol = document.getElementById('cholesterol').value;

if (!age || !gender || !bmi || !bp || !cholesterol) {
alert('Please fill in all fields');
return;
}

// Simple mock analysis (in a real app, this would call an API)
let risks = [];
let recommendations = [];

// BMI analysis
if (bmi > 30) {
risks.push('Obesity (High Risk)');
recommendations.push('Weight management program recommended');
} else if (bmi > 25) {
risks.push('Overweight (Moderate Risk)');
recommendations.push('Diet and exercise counseling suggested');
}

// Blood pressure analysis
const [systolic, diastolic] = bp.split('/').map(Number);
if (systolic >= 140 || diastolic >= 90) {
risks.push('Hypertension (High Risk)');
recommendations.push('Consult with cardiologist recommended');
} else if (systolic >= 130 || diastolic >= 85) {
risks.push('Elevated Blood Pressure (Moderate Risk)');
recommendations.push('Regular monitoring suggested');
}

// Cholesterol analysis
if (cholesterol > 240) {
risks.push('High Cholesterol (High Risk)');
recommendations.push('Lipid panel and dietary assessment recommended');
} else if (cholesterol > 200) {
risks.push('Borderline High Cholesterol (Moderate Risk)');
recommendations.push('Dietary modifications suggested');
}

// Age/gender specific risks
if (age > 50) {
risks.push('Age-related risk factors (Moderate Risk)');
recommendations.push('Annual comprehensive screening recommended');
}

if (gender === 'male' && age > 45) {
risks.push('Elevated cardiovascular risk (Moderate Risk)');
recommendations.push('Cardiac evaluation suggested');
} else if (gender === 'female' && age > 55) {
risks.push('Elevated cardiovascular risk (Moderate Risk)');
recommendations.push('Cardiac evaluation suggested');
}

// Display results
const resultDiv = document.getElementById('result-content');
const predictionDiv = document.getElementById('prediction-result');

if (risks.length === 0) {
resultDiv.innerHTML = `
<p><strong>Risk Assessment:</strong> Low overall risk based on provided data</p>
<p><strong>Recommendation:</strong> Continue with regular preventive care and healthy lifestyle</p>
`;
} else {
resultDiv.innerHTML = `
<p><strong>Identified Risk Factors:</strong></p>
<ul>
${risks.map(risk => `<li>${risk}</li>`).join('')}
</ul>
<p><strong>Clinical Recommendations:</strong></p>
<ul>
${recommendations.map(rec => `<li>${rec}</li>`).join('')}
</ul>
<p><em>Note: This is a simplified demo. Actual clinical decisions should be based on comprehensive evaluation.</em></p>
`;
}

predictionDiv.style.display = 'block';
}
</script>
